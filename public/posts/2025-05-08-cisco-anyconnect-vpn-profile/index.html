(WIP) God Mode Profile

<?xml version="1.0" encoding="UTF-8"?>
<!--
  God Mode AnyConnect profile
  All tags that possess a UserControllable attribute are present and set to "true".
  Replace CHANGEME placeholders with production values before deployment.
-->
<AnyConnectProfile xmlns="http://schemas.xmlsoap.org/encoding/"
                   xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance"
                   xsi:schemaLocation="http://schemas.xmlsoap.org/encoding/ AnyConnectProfile.xsd">
  <ClientInitialization>
    <!-- Start‑Before‑Logon -->
    <UseStartBeforeLogon UserControllable="true">true</UseStartBeforeLogon>

    <!-- Certificate handling -->
    <AutomaticCertSelection UserControllable="true">true</AutomaticCertSelection>
    <ShowPreConnectMessage>true</ShowPreConnectMessage>
    <CertificateStore>All</CertificateStore>
    <CertificateStoreMac>All</CertificateStoreMac>
    <CertificateStoreLinux>All</CertificateStoreLinux>
    <CertificateStoreOverride>true</CertificateStoreOverride>

    <!-- Proxy behaviour (all user‑tweakable) -->
    <ProxySettings UserControllable="true">Native
      <PublicProxyServerAddress UserControllable="true">CHANGEME.proxy.local:8080</PublicProxyServerAddress>
    </ProxySettings>
    <AllowLocalProxyConnections>true</AllowLocalProxyConnections>

    <!-- Connection workflow -->
    <AutoConnectOnStart UserControllable="true">true</AutoConnectOnStart>
    <MinimizeOnConnect UserControllable="true">true</MinimizeOnConnect>
    <LocalLanAccess UserControllable="true">true</LocalLanAccess>
    <DisableCaptivePortalDetection UserControllable="true">true</DisableCaptivePortalDetection>
    <AuthenticationTimeout>30</AuthenticationTimeout>

    <!-- Re‑connect logic -->
    <AutoReconnect UserControllable="true">true
      <AutoReconnectBehavior UserControllable="true">ReconnectAfterResume</AutoReconnectBehavior>
    </AutoReconnect>
    <SuspendOnConnectedStandby>true</SuspendOnConnectedStandby>

    <!-- Updates & tokens -->
    <AutoUpdate UserControllable="true">true</AutoUpdate>
    <RSASecurIDIntegration UserControllable="true">Automatic</RSASecurIDIntegration>

    <!-- OS‑specific policies (not user‑controllable in schema) -->
    <WindowsLogonEnforcement>SingleLocalLogon</WindowsLogonEnforcement>
    <LinuxLogonEnforcement>SingleLocalLogon</LinuxLogonEnforcement>
    <WindowsVPNEstablishment>AllowRemoteUsers</WindowsVPNEstablishment>
    <LinuxVPNEstablishment>AllowRemoteUsers</LinuxVPNEstablishment>

    <!-- Retain VPN on logoff (no UC attribute) -->
    <RetainVpnOnLogoff>true
      <UserEnforcement>AnyUser</UserEnforcement>
    </RetainVpnOnLogoff>

    <!-- PPP exclusion control -->
    <PPPExclusion UserControllable="true">Override
      <PPPExclusionServerIP UserControllable="true">10.0.0.1</PPPExclusionServerIP>
    </PPPExclusion>

    <!-- Scripting hooks -->
    <EnableScripting UserControllable="true">true
      <TerminateScriptOnNextEvent>true</TerminateScriptOnNextEvent>
      <EnablePostSBLOnConnectScript>true</EnablePostSBLOnConnectScript>
    </EnableScripting>

    <!-- Auto server selection -->
    <EnableAutomaticServerSelection UserControllable="true">true
      <AutoServerSelectionImprovement>25</AutoServerSelectionImprovement>
      <AutoServerSelectionSuspendTime>6</AutoServerSelectionSuspendTime>
    </EnableAutomaticServerSelection>

    <!-- Smartcard & token options -->
    <ClearSmartcardPin UserControllable="true">true</ClearSmartcardPin>
    <SafeWordSofTokenIntegration UserControllable="true">true</SafeWordSofTokenIntegration>

    <!-- Misc -->
    <IPProtocolSupport>IPv4,IPv6</IPProtocolSupport>
    <CaptivePortalRemediationBrowserFailover>true</CaptivePortalRemediationBrowserFailover>
    <AllowManualHostInput>true</AllowManualHostInput>
  </ClientInitialization>

  <!-- Headend list – obfuscated -->
  <ServerList>
    <HostEntry>
      <HostName>Primary VPN</HostName>
      <HostAddress>CHANGEME.vpn.example.com:443</HostAddress>
      <UserGroup>CHANGEME‑GROUP</UserGroup>
      <BackupServerList>
        <HostAddress>backup1.CHANGEME.example.com</HostAddress>
        <HostAddress>backup2.CHANGEME.example.com</HostAddress>
      </BackupServerList>
    </HostEntry>
  </ServerList>
</AnyConnectProfile>
